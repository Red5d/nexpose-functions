def NexposeLogin(consoleIP)
    require "io/console"

    print "Username: "
    username = $stdin.gets.chomp
    print "Password: "
    password = STDIN.noecho(&:gets).chomp
    puts ""

    # Create connection and login.
    nsc = Nexpose::Connection.new(consoleIP, username, password)
    nsc.login
    return nsc
end
